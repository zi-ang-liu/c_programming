    {
        printf("#%d\n", i + 1);
        for (j = 0; j < 3; j++)
        {

        }
    }
    //それぞれの平均値を計算
    for (i = 0; i < 3; i++)

#include <stdio.h>
void cmp(char *s1, int *r)
{
    char *s2 = "microwave oven\n";
    *r = 1;
    while (*s1 || *s2)
    {
        if (*s1 != *s2)
            return;
        s1++;
        s2++;
    }
    *r = 0;
}

    fgets(s, 80, stdin);

    else
        printf("Yes\n");
    return 0;
}

void reverse(char *s)
{
    int wk, n1 = 0, n2;
    n2 = strlen(s) - 1;
    while (n1 < n2)
    {
        wk = *(s + n1);
        *(s + n1) = *(s + n2);
        *(s + n2) = wk;
        ++n1;
        --n2;
    }
}

void repeat(char *strp, int n)
{
    int i;
    char *p = strp;
    for (i = 0; i < n; i++)
    {
        if (*strp == '\0')
            strp = p;
        printf("%c", *(strp++));
    }
    printf("\n");
}
